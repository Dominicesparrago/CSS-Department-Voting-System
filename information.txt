function lerpC(t){                              // map 0..1 -> teal ramp rgb
  t=Math.max(0,Math.min(1,t)); const f=t*(RAMP.length-1);
  const i=Math.floor(f), k=f-i, a=RAMP[i], b=RAMP[Math.min(i+1,RAMP.length-1)];
